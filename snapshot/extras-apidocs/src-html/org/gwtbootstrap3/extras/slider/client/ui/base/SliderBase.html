487    @Override
488    public void setValue(final T value, final boolean fireEvents) {
489
490        T oldValue = fireEvents ? getValue() : null;
491
492        if (isAttached()) {
493            setValue(getElement(), value);
494        } else {
495            String attrVal = (value == null) ? null : value.toString();
496            attributeMixin.setAttribute(SliderOption.VALUE.getDataAttribute(), attrVal);
497        }
498
499        if (fireEvents) {
500            T newValue = getValue();
501            ValueChangeEvent.fireIfNotEqual(this, oldValue, newValue);
502        }
503    }

702    private <E extends Enum<E>> E getEnumAttribute(SliderOption option, Class<E> clazz, E defaultValue) {
703        String value;
704        if (isAttached()) {
705            value = getStringAttribute(getElement(), option.getName());
706        } else {
707            value = attributeMixin.getAttribute(option.getDataAttribute());
708        }
709        try {
710            return Enum.valueOf(clazz, value);
711        } catch (Throwable e) {
712            return defaultValue;
713        }
714    }

716    private List<Double> getNumberArrayAttribute(SliderOption option, List<Double> defaultValue) {
717
718        // Get array attribute
719        JsArrayNumber array = null;
720        if (isAttached()) {
721            array = getNumberArrayAttribute(getElement(), option.getName());
722        } else {
723            String value = attributeMixin.getAttribute(option.getDataAttribute());
724            if (value != null && !value.isEmpty()) {
725                array = JsonUtils.safeEval(value);
726            }
727        }
728
729        // Attribute not set
730        if (array == null) {
731            return defaultValue;
732        }
733
734        // Put array to list
735        List<Double> list = new ArrayList<Double>(array.length());
736        for (int i = 0; i < array.length(); i++) {
737            list.add(array.get(i));
738        }
739        return list;
740    }
